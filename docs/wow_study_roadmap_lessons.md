## Step 1 — Data Shape in DB

The `studyPath` array is already generated by `/api/onboard/complete` via Claude Sonnet and stored in `students.memory.studyPath`. Make sure each node has this shape:

```typescript
// types/roadmap.ts
export interface RoadmapNode {
  id: string;              // uuid — e.g. "node_abc123"
  subject: string;         // MUST match subject name exactly — e.g. "Physics"
  topic: string;           // specific topic — e.g. "Photoelectric Effect"
  title: string;           // lesson title — e.g. "Functions"
  description: string;     // 1–2 sentence description of what this node covers
  weekNumber: number;      // which week this appears in
  orderInWeek: number;     // position within the week (1, 2, 3...)
  status: NodeStatus;      // see below
  estimatedMinutes: number; // e.g. 25
  difficulty: 'foundation' | 'core' | 'advanced';
}

export type NodeStatus =
  | 'completed'    // student finished this node
  | 'in_progress'  // currently working on this
  | 'up_next'      // next to do, unlocked
  | 'locked';      // not yet unlocked
```

## Step 2 — Backend: New Endpoints

### `GET /api/study-path/:studentId` (already planned — implement now)

```typescript
// routes/study-path.ts
router.get('/api/study-path/:studentId', async (req, res) => {
  const { studentId } = req.params;

  const student = await db.query(
    'SELECT * FROM students WHERE id = $1',
    [studentId]
  );

  if (!student.rows[0]) {
    return res.status(404).json({ error: 'Student not found' });
  }

  const memory = student.rows[0].memory;

  // Group roadmap nodes by subject
  const nodesBySubject: Record<string, RoadmapNode[]> = {};
  for (const node of memory.studyPath) {
    if (!nodesBySubject[node.subject]) nodesBySubject[node.subject] = [];
    nodesBySubject[node.subject].push(node);
  }

  // Sort each subject's nodes by weekNumber then orderInWeek
  for (const subject in nodesBySubject) {
    nodesBySubject[subject].sort((a, b) =>
      a.weekNumber !== b.weekNumber
        ? a.weekNumber - b.weekNumber
        : a.orderInWeek - b.orderInWeek
    );
  }

  res.json({
    studentId,
    name: student.rows[0].name,
    xp: student.rows[0].xp,
    streak: student.rows[0].streak,
    nodesBySubject,             // all subjects
    studyPath: memory.studyPath, // flat array too
    examDates: memory.examDates,
    goals: memory.goals,
    subjects: memory.subjects,
  });
});
```

### `GET /api/study-path/:studentId/:subject` (new — subject-specific)

```typescript
router.get('/api/study-path/:studentId/:subject', async (req, res) => {
  const { studentId, subject } = req.params;
  const subjectDecoded = decodeURIComponent(subject); // "Physics" not "physics"

  const student = await db.query(
    'SELECT * FROM students WHERE id = $1',
    [studentId]
  );

  const memory = student.rows[0].memory;

  // Case-insensitive match
  const nodes = memory.studyPath.filter(
    (n: RoadmapNode) => n.subject.toLowerCase() === subjectDecoded.toLowerCase()
  );

  nodes.sort((a: RoadmapNode, b: RoadmapNode) =>
    a.weekNumber !== b.weekNumber
      ? a.weekNumber - b.weekNumber
      : a.orderInWeek - b.orderInWeek
  );

  // Group by week
  const byWeek: Record<number, RoadmapNode[]> = {};
  for (const node of nodes) {
    if (!byWeek[node.weekNumber]) byWeek[node.weekNumber] = [];
    byWeek[node.weekNumber].push(node);
  }

  const totalWeeks = Math.max(...nodes.map((n: RoadmapNode) => n.weekNumber));
  const currentWeek = nodes.find(
    (n: RoadmapNode) => n.status === 'in_progress'
  )?.weekNumber ?? 1;

  res.json({
    subject: subjectDecoded,
    nodes,
    byWeek,
    currentWeek,
    totalWeeks,
    xp: student.rows[0].xp,
    streak: student.rows[0].streak,
    name: student.rows[0].name,
    examDate: memory.examDates?.find(
      (e: any) => e.subject.toLowerCase() === subjectDecoded.toLowerCase()
    )?.date ?? null,
    badges: memory.badges ?? [],
    studyHoursPerDay: memory.studyHoursPerDay,
  });
});
```

### `POST /api/lesson/content` (new — generates lesson for a node)

```typescript
router.post('/api/lesson/content', async (req, res) => {
  const { nodeId, studentId } = req.body;

  // Load student + find the node
  const student = await db.query(
    'SELECT * FROM students WHERE id = $1', [studentId]
  );
  const memory = student.rows[0].memory;
  const node = memory.studyPath.find((n: RoadmapNode) => n.id === nodeId);

  if (!node) return res.status(404).json({ error: 'Node not found' });

  const weakTopics = memory.subjects?.[node.subject]?.weak ?? [];
  const learningStyle = memory.learningStyle ?? 'examples';

  // Generate lesson with Claude Sonnet
  const response = await anthropic.messages.create({
    model: 'claude-sonnet-4-6',
    max_tokens: 2000,
    messages: [{
      role: 'user',
      content: `Generate an interactive lesson for a student.

Subject: ${node.subject}
Topic: ${node.topic}
Lesson title: ${node.title}
Description: ${node.description}
Student's weak areas: ${weakTopics.join(', ') || 'none identified'}
Learning style: ${learningStyle}
Difficulty: ${node.difficulty}

Return ONLY valid JSON matching this exact structure:
{
  "intro": "2-3 sentence engaging introduction to this topic",
  "keyPoints": [
    { "title": "string", "explanation": "2-3 sentences", "example": "concrete example" }
  ],
  "quiz": [
    {
      "question": "string",
      "options": ["A", "B", "C", "D"],
      "correctIndex": 0,
      "explanation": "why this answer is correct"
    }
  ],
  "summary": "1-2 sentence takeaway",
  "xpReward": 50
}

Rules:
- keyPoints: exactly 3 items
- quiz: exactly 3 questions
- All content MUST be specific to "${node.subject} — ${node.topic}"
- Do NOT include generic filler content
- If learningStyle is "examples", lead every explanation with a real-world example
- If student has weak areas, address them directly in the lesson`
    }]
  });

  const text = response.content[0].type === 'text' ? response.content[0].text : '';
  const clean = text.replace(/```json|```/g, '').trim();
  const lesson = JSON.parse(clean);

  res.json({ lesson, node });
});
```

### `POST /api/lesson/complete` (new — marks node done, awards XP)

```typescript
router.post('/api/lesson/complete', async (req, res) => {
  const { nodeId, studentId, score } = req.body;
  // score: number of correct quiz answers (0–3)

  const student = await db.query(
    'SELECT * FROM students WHERE id = $1', [studentId]
  );
  const memory = { ...student.rows[0].memory };

  // Update node status
  const nodeIndex = memory.studyPath.findIndex(
    (n: RoadmapNode) => n.id === nodeId
  );
  if (nodeIndex === -1) return res.status(404).json({ error: 'Node not found' });

  memory.studyPath[nodeIndex].status = 'completed';

  // Unlock next node
  const nextNode = memory.studyPath[nodeIndex + 1];
  if (nextNode && nextNode.status === 'locked') {
    memory.studyPath[nodeIndex + 1].status = 'up_next';
  }

  // XP: base 50 + bonus for quiz score
  const xpGained = 50 + (score * 15);
  const newXp = student.rows[0].xp + xpGained;

  // Update streak
  const today = new Date().toDateString();
  const lastActive = new Date(student.rows[0].last_active).toDateString();
  const yesterday = new Date(Date.now() - 86400000).toDateString();
  const newStreak = lastActive === yesterday
    ? student.rows[0].streak + 1
    : lastActive === today
      ? student.rows[0].streak
      : 1;

  // Save
  await db.query(
    `UPDATE students
     SET memory = $1, xp = $2, streak = $3, last_active = NOW()
     WHERE id = $4`,
    [JSON.stringify(memory), newXp, newStreak, studentId]
  );

  res.json({ xpGained, newXp, newStreak, nextNodeUnlocked: nextNode?.id ?? null });
});
```

import { useRouter } from 'next/navigation';

export function RoadmapNodeCard({ node, subject }) {
  const router = useRouter();
  const isClickable = node.status !== 'locked';

  const handleClick = () => {
    if (!isClickable) return;
    // Navigate to lesson page, passing nodeId
    router.push(`/lesson/${node.id}`);
  };

  return (
    <div
      className={`roadmap-node roadmap-node--${node.status}`}
      onClick={handleClick}
      role={isClickable ? 'button' : undefined}
      tabIndex={isClickable ? 0 : undefined}
      onKeyDown={e => e.key === 'Enter' && handleClick()}
      title={node.status === 'locked' ? 'Complete previous nodes first' : ''}
    >
      {/* Subject + Topic tag */}
      <span className="node-tag">
        {node.subject.toUpperCase()} — {node.topic.toUpperCase()}
      </span>

      {/* Lesson title */}
      <h3 className="node-title">{node.title}</h3>

      {/* Status badge */}
      <StatusBadge status={node.status} />

      {/* Hover overlay for clickable nodes */}
      {isClickable && (
        <div className="node-hover-overlay">
          <span>Start lesson →</span>
        </div>
      )}
    </div>
  );
}

function StatusBadge({ status }) {
  const labels = {
    in_progress: 'In progress',
    up_next:     'Up next',
    locked:      'Locked',
    completed:   'Completed',
  };
  return (
    <span className={`status-badge status-badge--${status}`}>
      {labels[status]}
    </span>
  );
}

## Step 4 — Lesson Page

### `app/lesson/[nodeId]/page.tsx`

```tsx
'use client';

import { useEffect, useState } from 'react';
import { useParams, useRouter } from 'next/navigation';

const API = 'https://hackatonhoiv-production.up.railway.app';

type LessonPhase = 'intro' | 'keypoints' | 'quiz' | 'summary';

export default function LessonPage() {
  const { nodeId } = useParams<{ nodeId: string }>();
  const router = useRouter();

  const [lesson, setLesson] = useState<LessonContent | null>(null);
  const [node, setNode] = useState<RoadmapNode | null>(null);
  const [loading, setLoading] = useState(true);
  const [phase, setPhase] = useState<LessonPhase>('intro');

  // Quiz state
  const [currentQ, setCurrentQ] = useState(0);
  const [selected, setSelected] = useState<number | null>(null);
  const [answered, setAnswered] = useState(false);
  const [score, setScore] = useState(0);
  const [quizDone, setQuizDone] = useState(false);

  // Completion state
  const [completing, setCompleting] = useState(false);
  const [xpGained, setXpGained] = useState(0);

  useEffect(() => {
    const studentId = localStorage.getItem('studentId');
    if (!studentId) { router.push('/onboarding'); return; }

    fetch(`${API}/api/lesson/content`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ nodeId, studentId }),
    })
      .then(r => r.json())
      .then(d => {
        setLesson(d.lesson);
        setNode(d.node);
        setLoading(false);
      });
  }, [nodeId]);

  const handleQuizAnswer = (index: number) => {
    if (answered) return;
    setSelected(index);
    setAnswered(true);
    if (index === lesson!.quiz[currentQ].correctIndex) {
      setScore(prev => prev + 1);
    }
  };

  const handleNextQuestion = () => {
    if (currentQ < lesson!.quiz.length - 1) {
      setCurrentQ(prev => prev + 1);
      setSelected(null);
      setAnswered(false);
    } else {
      setQuizDone(true);
    }
  };

  const handleComplete = async () => {
    setCompleting(true);
    const studentId = localStorage.getItem('studentId')!;

    const res = await fetch(`${API}/api/lesson/complete`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ nodeId, studentId, score }),
    });
    const data = await res.json();

    setXpGained(data.xpGained);
    setPhase('summary');
    setCompleting(false);
  };

  if (loading) return <LessonSkeleton />;
  if (!lesson || !node) return <div>Lesson not found.</div>;

  return (
    <div className="lesson-page">
      <LessonHeader node={node} phase={phase} onBack={() => router.back()} />

      <div className="lesson-content">
        {phase === 'intro' && (
          <IntroPhase
            lesson={lesson}
            node={node}
            onContinue={() => setPhase('keypoints')}
          />
        )}
        {phase === 'keypoints' && (
          <KeyPointsPhase
            keyPoints={lesson.keyPoints}
            onContinue={() => setPhase('quiz')}
          />
        )}
        {phase === 'quiz' && !quizDone && (
          <QuizPhase
            question={lesson.quiz[currentQ]}
            questionIndex={currentQ}
            total={lesson.quiz.length}
            selected={selected}
            answered={answered}
            score={score}
            onSelect={handleQuizAnswer}
            onNext={handleNextQuestion}
          />
        )}
        {phase === 'quiz' && quizDone && (
          <QuizResults
            score={score}
            total={lesson.quiz.length}
            onComplete={handleComplete}
            completing={completing}
          />
        )}
        {phase === 'summary' && (
          <SummaryPhase
            summary={lesson.summary}
            xpGained={xpGained}
            score={score}
            total={lesson.quiz.length}
            onReturn={() => router.push(`/dashboard/${encodeURIComponent(node.subject)}`)}
          />
        )}
      </div>
    </div>
  );
}
```

## Full Route Map

```
/dashboard                          ← subject card grid
/dashboard/[subject]                ← subject detail (roadmap + today panel)
/lesson/[nodeId]                    ← interactive lesson (4 phases)
```

---

## Key Rules

1. **Roadmap nodes are ALWAYS fetched from DB** — never hardcoded. Subject detail page only renders nodes where `node.subject === selectedSubject`.

2. **Lesson content is generated per-node by Claude** — it uses `node.subject`, `node.topic`, and `node.title` to generate fully subject-specific content. History gets WW1 lessons. Physics gets photoelectric effect. Never mix.

3. **Node unlocking is sequential** — completing a node calls `/api/lesson/complete` which sets next node from `locked` → `up_next` and saves to DB.

4. **XP is real** — it's stored in Postgres and returned on every `/api/study-path` call. The XP bar on the subject detail page always reflects the live value.

5. **All navigation returns to the correct subject** — after completing a lesson, `router.push('/dashboard/${node.subject}')` ensures the student lands back on their Physics roadmap, not a generic dashboard.
